//////Configurar eigrp///////

router eigrp 1
	network x.x.x.x x.x.x.x (wildcard)
	passive-interface ___ (interfaz hacia lan)
	no auto-summary
	redistribute static //Propagar ruta estática
	redistribute ospf 1
	
	///opcional///
		ip bandwidth-percent eigrp 1 75
		bandwidth 1075
		ip hello-interval eigrp 1 60
		ip hold-time eigrp 1 180

//////Configurar OSPF///////
	
router ospf 1
	router-id x.x.x.x (No es una ip, es una id con forma de ip.. podria incluso ser 0.0.0.1, 0.0.0.2 y asi...)
	network x.x.x.x x.x.x.x (wildcard) area 0
	passive-interface ___
	redistribute eigrp 1
	default-info origin
	////////////
	ipv6 router ospf 1

	int __
	ipv6 ospf 1 area 0
